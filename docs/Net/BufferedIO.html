def initialize_with_webmock(io, debug_output = nil)
  @read_timeout = 60
  @rbuf = ''
  @debug_output = debug_output

  @io = case io
  when Socket, OpenSSL::SSL::SSLSocket, IO
    io
  when String
    StringIO.new(io)
  end
  raise "Unable to create local socket" unless @io
end
