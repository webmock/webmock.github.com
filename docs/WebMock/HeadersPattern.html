def matches?(headers)
  if empty_headers?(@pattern)
    empty_headers?(headers)
  else
    return false if empty_headers?(headers)
    @pattern.each do |key, value|
      return false unless headers.has_key?(key) && value === headers[key]
    end
    true
  end
end
